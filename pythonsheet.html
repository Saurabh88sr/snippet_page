    Factorial

    
                def factorial(n):
    if n==1:
        return 1
    else:
        return (n*factorial(n-1))

num= int(input("Enter a Number: "))
if num<0:
    print("Factorial dose not exist for negative number")
elif num==0:
    print("The factorial is 0 is 1")

else:
    print("number: ", num)
    print("The Factorial is", factorial(num))
